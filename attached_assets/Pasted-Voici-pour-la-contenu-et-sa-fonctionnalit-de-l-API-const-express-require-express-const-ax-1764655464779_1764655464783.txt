const express = require('express');
const axios = require('axios');
const cheerio = require('cheerio');

const app = express();
const PORT = process.env.PORT || 5000;

app.use(express.json());

app.get('/recherche', async (req, res) => {
  try {
    const { video = 'Vicki', page = '0' } = req.query;
    
    // Construct URL based on page number
    let url;
    if (page === '0' || page === '1') {
      url = `https://www.xnxx.com/search/${encodeURIComponent(video)}`;
    } else {
      url = `https://www.xnxx.com/search/${encodeURIComponent(video)}/${parseInt(page) - 1}`;
    }

    // Fetch the HTML page
    const response = await axios.get(url, {
      headers: {
        'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36'
      }
    });

    const html = response.data;
    const $ = cheerio.load(html);
    
    const videos = [];

    // Parse video thumbnails from the mozaique div
    $('.mozaique .thumb-block').each((index, element) => {
      const $element = $(element);
      
      // Extract title
      const titleLink = $element.find('.thumb-under p a');
      const titre = titleLink.attr('title') || titleLink.text().trim();
      
      // Extract image URL in HD quality
      const img = $element.find('.thumb img');
      // Priority order: if (HD) > data-sfwthumb > data-src
      let image_url = img.attr('if') || img.attr('data-sfwthumb') || img.attr('data-src') || img.attr('src');
      
      // If we have thumbs169xnxx URL, convert to thumbs169lll (HD)
      if (image_url && image_url.includes('thumbs169xnxx')) {
        image_url = image_url.replace('thumbs169xnxx', 'thumbs169lll');
      }
      
      // Extract video URL (link to video page)
      const videoPageUrl = titleLink.attr('href');
      const video_url = videoPageUrl ? `https://www.xnxx.com${videoPageUrl}` : '';
      
      // Extract duration
      const duration = $element.find('.thumb-under .metadata').text().match(/\d+min|\d+sec/)?.[0] || '';
      
      // Extract views
      const viewsText = $element.find('.thumb-under .metadata .right').text().trim();
      const views = viewsText.split(' ')[0] || '';

      if (titre && image_url && video_url) {
        videos.push({
          titre: titre,
          image_url: image_url,
          video_url: video_url,
          duration: duration,
          views: views
        });
      }
    });

    res.json({
      success: true,
      query: video,
      page: parseInt(page) || 1,
      total_results: videos.length,
      videos: videos
    });

  } catch (error) {
    console.error('Error:', error.message);
    res.status(500).json({
      success: false,
      error: 'Failed to scrape data',
      message: error.message
    });
  }
});

app.get('/download', async (req, res) => {
  try {
    const { video } = req.query;
    
    if (!video) {
      return res.status(400).json({
        success: false,
        error: 'Missing video URL parameter',
        usage: '/download?video=https://www.xnxx.com/video-xxxxx/title'
      });
    }

    const response = await axios.get(video, {
      headers: {
        'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36'
      }
    });

    const html = response.data;
    
    let downloadUrls = {
      low: null,
      high: null,
      hls: null
    };

    const lowMatch = html.match(/html5player\.setVideoUrlLow\('([^']+)'\)/);
    if (lowMatch) {
      downloadUrls.low = lowMatch[1];
    }

    const highMatch = html.match(/html5player\.setVideoUrlHigh\('([^']+)'\)/);
    if (highMatch) {
      downloadUrls.high = highMatch[1];
    }

    const hlsMatch = html.match(/html5player\.setVideoHLS\('([^']+)'\)/);
    if (hlsMatch) {
      downloadUrls.hls = hlsMatch[1];
    }

    const $ = cheerio.load(html);
    const title = $('title').text().replace(' - XNXX.COM', '').trim() || 'video';

    const bestQuality = downloadUrls.low || downloadUrls.high;

    if (!bestQuality) {
      return res.status(404).json({
        success: false,
        error: 'Could not find download URL',
        message: 'No video URL found on this page'
      });
    }

    const safeTitle = title.replace(/[^a-zA-Z0-9]/g, '_').substring(0, 50);
    const filename = `${safeTitle}.mp4`;

    const videoResponse = await axios({
      method: 'GET',
      url: bestQuality,
      responseType: 'stream',
      headers: {
        'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36'
      }
    });

    res.setHeader('Content-Type', 'video/mp4');
    res.setHeader('Content-Disposition', `attachment; filename="${filename}"`);
    
    if (videoResponse.headers['content-length']) {
      res.setHeader('Content-Length', videoResponse.headers['content-length']);
    }

    videoResponse.data.pipe(res);

  } catch (error) {
    console.error('Download Error:', error.message);
    res.status(500).json({
      success: false,
      error: 'Failed to get download URL',
      message: error.message
    });
  }
});

app.get('/', (req, res) => {
  res.json({

      download: '/download?video=URL_VIDEO'

  });
});

app.listen(PORT, '0.0.0.0', () => {
  console.log(`Server running on port ${PORT}`);
});
